95  // This table is indexed using the top 5 bits of the first byte
96  0, 4, 3, 3, 2, 2, 2, 2, 0, 0, 0, 0, 0, 0, 0, 0,

109  // Once ingesting, subsequent bytes MUST have the top two bits be 10.
110  // These tables guarantee a legal code path or error.
111  static const void *dblabel[] = { &&err, &&err, &&db, &&err };
112  static const void *dclabel[] = { &&err, &&err, &&dc, &&err };
